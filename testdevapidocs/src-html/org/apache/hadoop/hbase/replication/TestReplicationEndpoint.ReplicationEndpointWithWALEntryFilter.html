075  @Before
076  public void setup() throws Exception {
077    ReplicationEndpointForTest.contructedCount.set(0);
078    ReplicationEndpointForTest.startedCount.set(0);
079    ReplicationEndpointForTest.replicateCount.set(0);
080    ReplicationEndpointReturningFalse.replicated.set(false);
081    ReplicationEndpointForTest.lastEntries = null;
082    final List<RegionServerThread> rsThreads =
083        utility1.getMiniHBaseCluster().getRegionServerThreads();
084    for (RegionServerThread rs : rsThreads) {
085      utility1.getHBaseAdmin().rollWALWriter(rs.getRegionServer().getServerName());
086    }
087    // Wait for  all log roll to finish
088    utility1.waitFor(3000, new Waiter.ExplainingPredicate<Exception>() {
089      @Override
090      public boolean evaluate() throws Exception {
091        for (RegionServerThread rs : rsThreads) {
092          if (!rs.getRegionServer().walRollRequestFinished()) {
093            return false;
094          }
095        }
096        return true;
097      }
098
099      @Override
100      public String explainFailure() throws Exception {
101        List<String> logRollInProgressRsList = new ArrayList<String>();
102        for (RegionServerThread rs : rsThreads) {
103          if (!rs.getRegionServer().walRollRequestFinished()) {
104            logRollInProgressRsList.add(rs.getRegionServer().toString());
105          }
106        }
107        return "Still waiting for log roll on regionservers: " + logRollInProgressRsList;
108      }
109    });
110  }

112  @Test (timeout=120000)
113  public void testCustomReplicationEndpoint() throws Exception {
114    // test installing a custom replication endpoint other than the default one.
115    admin.addPeer("testCustomReplicationEndpoint",
116      new ReplicationPeerConfig().setClusterKey(ZKUtil.getZooKeeperClusterKey(conf1))
117        .setReplicationEndpointImpl(ReplicationEndpointForTest.class.getName()), null);
118
119    // check whether the class has been constructed and started
120    Waiter.waitFor(conf1, 60000, new Waiter.Predicate<Exception>() {
121      @Override
122      public boolean evaluate() throws Exception {
123        return ReplicationEndpointForTest.contructedCount.get() >= numRegionServers;
124      }
125    });
126
127    Waiter.waitFor(conf1, 60000, new Waiter.Predicate<Exception>() {
128      @Override
129      public boolean evaluate() throws Exception {
130        return ReplicationEndpointForTest.startedCount.get() >= numRegionServers;
131      }
132    });
133
134    Assert.assertEquals(0, ReplicationEndpointForTest.replicateCount.get());
135
136    // now replicate some data.
137    doPut(Bytes.toBytes("row42"));
138
139    Waiter.waitFor(conf1, 60000, new Waiter.Predicate<Exception>() {
140      @Override
141      public boolean evaluate() throws Exception {
142        return ReplicationEndpointForTest.replicateCount.get() >= 1;
143      }
144    });
145
146    doAssert(Bytes.toBytes("row42"));
147
148    admin.removePeer("testCustomReplicationEndpoint");
149  }

151  @Test (timeout=120000)
152  public void testReplicationEndpointReturnsFalseOnReplicate() throws Exception {
153    Assert.assertEquals(0, ReplicationEndpointForTest.replicateCount.get());
154    Assert.assertTrue(!ReplicationEndpointReturningFalse.replicated.get());
155    int peerCount = admin.getPeersCount();
156    final String id = "testReplicationEndpointReturnsFalseOnReplicate";
157    admin.addPeer(id,
158      new ReplicationPeerConfig().setClusterKey(ZKUtil.getZooKeeperClusterKey(conf1))
159        .setReplicationEndpointImpl(ReplicationEndpointReturningFalse.class.getName()), null);
160    // This test is flakey and then there is so much stuff flying around in here its, hard to
161    // debug.  Peer needs to be up for the edit to make it across. This wait on
162    // peer count seems to be a hack that has us not progress till peer is up.
163    if (admin.getPeersCount() <= peerCount) {
164      LOG.info("Waiting on peercount to go up from " + peerCount);
165      Threads.sleep(100);
166    }
167    // now replicate some data
168    doPut(row);
169
170    Waiter.waitFor(conf1, 60000, new Waiter.Predicate<Exception>() {
171      @Override
172      public boolean evaluate() throws Exception {
173        // Looks like replication endpoint returns false unless we put more than 10 edits. We
174        // only send over one edit.
175        int count = ReplicationEndpointForTest.replicateCount.get();
176        LOG.info("count=" + count);
177        return ReplicationEndpointReturningFalse.replicated.get();
178      }
179    });
180    if (ReplicationEndpointReturningFalse.ex.get() != null) {
181      throw ReplicationEndpointReturningFalse.ex.get();
182    }
183
184    admin.removePeer("testReplicationEndpointReturnsFalseOnReplicate");
185  }

207    admin.removePeer("testWALEntryFilterFromReplicationEndpoint");

211  private void doPut(byte[] row) throws IOException {
212    try (Connection connection = ConnectionFactory.createConnection(conf1)) {
213      doPut(connection, row);
214    }
215  }

217  private void doPut(final Connection connection, final byte [] row) throws IOException {
218    try (Table t = connection.getTable(tableName)) {
219      Put put = new Put(row);
220      put.addColumn(famName, row, row);
221      t.put(put);
222    }
223  }
224
225  private static void doAssert(byte[] row) throws Exception {
226    if (ReplicationEndpointForTest.lastEntries == null) {
227      return; // first call
228    }
229    Assert.assertEquals(1, ReplicationEndpointForTest.lastEntries.size());
230    List<Cell> cells = ReplicationEndpointForTest.lastEntries.get(0).getEdit().getCells();
231    Assert.assertEquals(1, cells.size());
232    Assert.assertTrue(Bytes.equals(cells.get(0).getRowArray(), cells.get(0).getRowOffset(),
233      cells.get(0).getRowLength(), row, 0, row.length));
234  }

236  public static class ReplicationEndpointForTest extends BaseReplicationEndpoint {
237    static UUID uuid = UUID.randomUUID();
238    static AtomicInteger contructedCount = new AtomicInteger();
239    static AtomicInteger startedCount = new AtomicInteger();
240    static AtomicInteger stoppedCount = new AtomicInteger();
241    static AtomicInteger replicateCount = new AtomicInteger();
242    static volatile List<Entry> lastEntries = null;
243
244    public ReplicationEndpointForTest() {
245      contructedCount.incrementAndGet();
246    }
247
248    @Override
249    public UUID getPeerUUID() {
250      return uuid;
251    }
252
253    @Override
254    public boolean replicate(ReplicateContext replicateContext) {
255      replicateCount.incrementAndGet();
256      lastEntries = replicateContext.entries;
257      return true;
258    }
259
260    @Override
261    protected void doStart() {
262      startedCount.incrementAndGet();
263      notifyStarted();
264    }
265
266    @Override
267    protected void doStop() {
268      stoppedCount.incrementAndGet();
269      notifyStopped();
270    }
271  }
272
273  public static class ReplicationEndpointReturningFalse extends ReplicationEndpointForTest {
274    static int COUNT = 10;
275    static AtomicReference<Exception> ex = new AtomicReference<Exception>(null);
276    static AtomicBoolean replicated = new AtomicBoolean(false);
277    @Override
278    public boolean replicate(ReplicateContext replicateContext) {
279      try {
280        // check row
281        doAssert(row);
282      } catch (Exception e) {
283        ex.set(e);
284      }
285
286      super.replicate(replicateContext);
287      LOG.info("Replicated " + row + ", count=" + replicateCount.get());
288
289      replicated.set(replicateCount.get() > COUNT); // first 10 times, we return false
290      return replicated.get();
291    }
292  }
293
294  // return a WALEntry filter which only accepts "row", but not other rows
295  public static class ReplicationEndpointWithWALEntryFilter extends ReplicationEndpointForTest {
296    static AtomicReference<Exception> ex = new AtomicReference<Exception>(null);
297
298    @Override
299    public boolean replicate(ReplicateContext replicateContext) {
300      try {
301        super.replicate(replicateContext);
302        doAssert(row);
303      } catch (Exception e) {
304        ex.set(e);
305      }
306      return true;
307    }
308
309    @Override
310    public WALEntryFilter getWALEntryfilter() {
311      return new ChainWALEntryFilter(super.getWALEntryfilter(), new WALEntryFilter() {
312        @Override
313        public Entry filter(Entry entry) {
314          ArrayList<Cell> cells = entry.getEdit().getCells();
315          int size = cells.size();
316          for (int i = size-1; i >= 0; i--) {
317            Cell cell = cells.get(i);
318            if (!Bytes.equals(cell.getRowArray(), cell.getRowOffset(), cell.getRowLength(),
319              row, 0, row.length)) {
320              cells.remove(i);
321            }
322          }
323          return entry;
324        }
325      });
326    }
