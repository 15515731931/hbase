019package org.apache.hadoop.hbase.regionserver;
020
021import java.io.IOException;
022import java.util.ArrayList;
023import java.util.Collection;
024import java.util.Collections;
025import java.util.Iterator;
026import java.util.List;
027
028import org.apache.commons.logging.Log;
029import org.apache.commons.logging.LogFactory;
030import org.apache.hadoop.conf.Configuration;
031import org.apache.hadoop.hbase.Cell;
032import org.apache.hadoop.hbase.CellComparator;
033import org.apache.hadoop.hbase.KeyValue;
034import org.apache.hadoop.hbase.classification.InterfaceAudience;
035import org.apache.hadoop.hbase.regionserver.compactions.CompactionConfiguration;
036
037import com.google.common.collect.ImmutableCollection;
038import com.google.common.collect.ImmutableList;
039import com.google.common.collect.Lists;
040
041/**
042 * Default implementation of StoreFileManager. Not thread-safe.
043 */
044@InterfaceAudience.Private
045class DefaultStoreFileManager implements StoreFileManager {
046  private static final Log LOG = LogFactory.getLog(DefaultStoreFileManager.class);
047
048  private final CellComparator kvComparator;
049  private final CompactionConfiguration comConf;
050  private final int blockingFileCount;
051
052  /**
053   * List of store files inside this store. This is an immutable list that
054   * is atomically replaced when its contents change.
055   */
056  private volatile ImmutableList<StoreFile> storefiles = null;
057
058  public DefaultStoreFileManager(CellComparator kvComparator, Configuration conf,
059      CompactionConfiguration comConf) {
060    this.kvComparator = kvComparator;
061    this.comConf = comConf;
062    this.blockingFileCount =
063        conf.getInt(HStore.BLOCKING_STOREFILES_KEY, HStore.DEFAULT_BLOCKING_STOREFILE_COUNT);
064  }
065
066  @Override
067  public void loadFiles(List<StoreFile> storeFiles) {
068    sortAndSetStoreFiles(storeFiles);
069  }
070
071  @Override
072  public final Collection<StoreFile> getStorefiles() {
073    return storefiles;
074  }
075
076  @Override
077  public void insertNewFiles(Collection<StoreFile> sfs) throws IOException {
078    ArrayList<StoreFile> newFiles = new ArrayList<StoreFile>(storefiles);
079    newFiles.addAll(sfs);
080    sortAndSetStoreFiles(newFiles);
081  }
082
083  @Override
084  public ImmutableCollection<StoreFile> clearFiles() {
085    ImmutableList<StoreFile> result = storefiles;
086    storefiles = ImmutableList.of();
087    return result;
088  }
089
090  @Override
091  public final int getStorefileCount() {
092    return storefiles.size();
093  }
094
095  @Override
096  public void addCompactionResults(
097    Collection<StoreFile> compactedFiles, Collection<StoreFile> results) {
098    ArrayList<StoreFile> newStoreFiles = Lists.newArrayList(storefiles);
099    newStoreFiles.removeAll(compactedFiles);
100    if (!results.isEmpty()) {
101      newStoreFiles.addAll(results);
102    }
103    sortAndSetStoreFiles(newStoreFiles);
104  }
105
106  @Override
107  public final Iterator<StoreFile> getCandidateFilesForRowKeyBefore(final KeyValue targetKey) {
108    return new ArrayList<StoreFile>(Lists.reverse(this.storefiles)).iterator();
109  }
110
111  @Override
112  public Iterator<StoreFile> updateCandidateFilesForRowKeyBefore(
113      Iterator<StoreFile> candidateFiles, final KeyValue targetKey, final Cell candidate) {
114    // Default store has nothing useful to do here.
115    // TODO: move this comment when implementing Level:
116    // Level store can trim the list by range, removing all the files which cannot have
117    // any useful candidates less than "candidate".
118    return candidateFiles;
119  }
120
121  @Override
122  public final byte[] getSplitPoint() throws IOException {
123    if (this.storefiles.isEmpty()) {
124      return null;
125    }
126    return StoreUtils.getLargestFile(this.storefiles).getFileSplitPoint(this.kvComparator);
127  }
128
129  @Override
130  public final Collection<StoreFile> getFilesForScanOrGet(boolean isGet,
131      byte[] startRow, byte[] stopRow) {
132    // We cannot provide any useful input and already have the files sorted by seqNum.
133    return getStorefiles();
134  }
135
136  @Override
137  public int getStoreCompactionPriority() {
138    int priority = blockingFileCount - storefiles.size();
139    return (priority == HStore.PRIORITY_USER) ? priority + 1 : priority;
140  }
141
142  @Override
143  public Collection<StoreFile> getUnneededFiles(long maxTs, List<StoreFile> filesCompacting) {
144    Collection<StoreFile> expiredStoreFiles = null;
145    ImmutableList<StoreFile> files = storefiles;
146    // 1) We can never get rid of the last file which has the maximum seqid.
147    // 2) Files that are not the latest can't become one due to (1), so the rest are fair game.
148    for (int i = 0; i < files.size() - 1; ++i) {
149      StoreFile sf = files.get(i);
150      long fileTs = sf.getReader().getMaxTimestamp();
151      if (fileTs < maxTs && !filesCompacting.contains(sf)) {
152        LOG.info("Found an expired store file: " + sf.getPath()
153            + " whose maxTimeStamp is " + fileTs + ", which is below " + maxTs);
154        if (expiredStoreFiles == null) {
155          expiredStoreFiles = new ArrayList<StoreFile>();
156        }
157        expiredStoreFiles.add(sf);
158      }
159    }
160    return expiredStoreFiles;
161  }
162
163  private void sortAndSetStoreFiles(List<StoreFile> storeFiles) {
164    Collections.sort(storeFiles, StoreFile.Comparators.SEQ_ID);
165    storefiles = ImmutableList.copyOf(storeFiles);
166  }
167
168  @Override
169  public double getCompactionPressure() {
170    int storefileCount = getStorefileCount();
171    int minFilesToCompact = comConf.getMinFilesToCompact();
172    if (storefileCount <= minFilesToCompact) {
173      return 0.0;
174    }
175    return (double) (storefileCount - minFilesToCompact) / (blockingFileCount - minFilesToCompact);
176  }
177}
